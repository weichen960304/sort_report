📝 JavaScript 實現

function quickSort(arr, low = 0, high = arr.length - 1) {
    if (low < high) {
        let pi = partition(arr, low, high);
        
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
    return arr;
}

function partition(arr, low, high) {
    let pivot = arr[high];
    let i = low - 1;
    
    for (let j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }
    
    [arr[i + 1], arr[high]] = [arr[high], arr[i + 1]];
    return i + 1;
}

📚 操作原理

快速排序法同樣使用「分而治之」的概念。它會在陣列中挑選一個基準值（Pivot），然後將陣列重新排列：所有比基準值小的元素放到基準值左邊，比基準值大的放到右邊。這個動作稱為分區（Partition）。接著對左右兩個子陣列遞歸執行相同的操作，直到子陣列長度為 1 或 0。

主要步驟如下：

  1. 從陣列中選擇一個基準值（通常是最右邊的元素）。
  2. 將陣列重新排列：比基準值小的元素放左邊，大的放右邊。
  3. 對左邊的子陣列和右邊的子陣列重複步驟 1-2。
  4. 直到所有子陣列都只剩一個元素為止。

⏱️ 時間與空間複雜度

快速排序法的時間與空間複雜度如下：

  • 時間複雜度：
    • 最佳時間複雜度：$O(n \log n)$（每次都能均勻分割陣列）
    • 平均時間複雜度：$O(n \log n)$
    • 最壞時間複雜度：$O(n^2)$（當資料已排序且選擇到極端值作為 Pivot 時）
  • 空間複雜度：$O(\log n)$（遞歸呼叫的堆疊空間，最壞情況下為 $O(n)$）

📋 完整程式碼

以下是使用 JavaScript 實現快速排序法的完整程式碼：

function quickSort(arr, low = 0, high = arr.length - 1) {
    if (low < high) {
        let pi = partition(arr, low, high);
        
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
    return arr;
}

function partition(arr, low, high) {
    let pivot = arr[high];
    let i = low - 1;
    
    for (let j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            // 交換元素
            let temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    
    // 將基準值放到正確位置
    let temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}

let array = [64, 34, 25, 12, 22, 11, 90];
console.log("排序前的陣列: ", array);
let sortedArray = quickSort(array);
console.log("排序後的陣列: ", sortedArray);
